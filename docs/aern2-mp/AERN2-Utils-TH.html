declForTypes :: [Q Type] -> (Q Type -> Q [Dec]) -> Q [Dec] Source #

A toy example of use:

class HasTT t where
  type TT t
  getTT :: t -> TT t

$(declForTypes
  [[t| Integer |], [t| Int |], [t| Rational |]]
  ( t -> [d|
    instance HasTT $t where
      type TT $t = ()
      getTT _ = ()
  |]))
